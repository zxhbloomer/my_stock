## 📋 KISS原则7问评估

### 1. 这是个真问题还是臆想出来的？

**✅ 真问题**

- **现状**: 项目使用Alpha158(158因子) + 自定义63因子 = 共221个因子
- **痛点**: 因子库相对有限，RD-Agent项目有375个AI生成的量化因子未被利用
- **真实需求**: 扩展因子池，通过IC分析筛选出更多有效因子，提升策略表现

**证据**:
- RD-Agent workspace包含610个factor.py文件，去重后375个唯一因子
- 现有factors/alpha_factors.py仅63个自定义因子
- IC分析显示因子质量参差不齐，需要更多高质量因子备选

### 2. 有更简单的方法吗？

**✅ 已选择最简方案**

**备选方案对比**:

| 方案 | 复杂度 | 工作量 | 风险 | 选择 |
|------|--------|--------|------|------|
| A. 迁移所有610个因子 | 高 | 40-80小时 | 高（代码膨胀） | ❌ 过度设计 |
| B. 自定义FactorHandler | 中 | 20-30小时 | 中（需适配数据格式） | ❌ 不必要 |
| C. 转换50-80个为Qlib表达式 | 低 | 6-10小时 | 低（纯新增） | ✅ **最简方案** |

**为什么方案C最简单**:
1. **复用现有架构**: 直接使用factors/alpha_factors.py的模式，零学习成本
2. **纯新增文件**: 不修改任何现有代码，零破坏风险
3. **智能筛选**: 只迁移高质量因子，避免代码膨胀
4. **即插即用**: 一行代码即可启用/禁用RD-Agent因子

### 文件设计：factors/rdagent_factors.py

**结构**:
```python
"""
RD-Agent因子库
从RD-Agent项目迁移的AI生成因子，转换为Qlib表达式格式
来源: https://github.com/microsoft/RD-Agent
"""


class RDAgentFactors:
    """RD-Agent AI生成因子（Qlib表达式格式）"""

    @staticmethod
    def get_price_momentum_features():
        """价格动量因子（12个）"""
        return [
            # 多周期动量
            "$close / Ref($close, 1) - 1",                    # 1日动量 (daily_return)
            "$close / Ref($close, 2) - 1",                    # 2日动量
            "$close / Ref($close, 3) - 1",                    # 3日动量
            "$close / Ref($close, 5) - 1",                    # 5日动量
            "$close / Ref($close, 10) - 1",                   # 10日动量
            "$close / Ref($close, 20) - 1",                   # 20日动量

            # 价格位置
            "($close - Min($low, 5)) / (Max($high, 5) - Min($low, 5))",    # 5日归一化位置
            "($close - Min($low, 10)) / (Max($high, 10) - Min($low, 10))", # 10日归一化位置
            "($close - Min($low, 20)) / (Max($high, 20) - Min($low, 20))", # 20日归一化位置

            # 价格比率
            "$close / $open",                                 # 收盘/开盘比
            "($close + $open) / 2 / $close",                 # 中点价格比
            "($close - $open) / ($high - $low)",             # 收盘位置
        ]

    @staticmethod
    def get_volume_features():
        """成交量因子（10个）"""
        return [
            # 成交量比率
            "$volume / Ref($volume, 1)",                      # 量比（1日）
            "$volume / Mean($volume, 5)",                     # 5日平均量比
            "$volume / Mean($volume, 10)",                    # 10日平均量比
            "$volume / Mean($volume, 20)",                    # 20日平均量比

            # 成交量标准化
            "($volume - Mean($volume, 10)) / Std($volume, 10)",  # 10日量标准化偏离
            "($volume - Mean($volume, 20)) / Std($volume, 20)",  # 20日量标准化偏离

            # 成交量动量
            "Mean($volume, 5) / Mean($volume, 20)",          # 短期/长期量比
            "Log($volume / Mean($volume, 5))",                # 对数量比

            # 成交量波动
            "Std($volume, 10) / Mean($volume, 10)",          # 10日量波动率
            "Std($volume, 20) / Mean($volume, 20)",          # 20日量波动率
        ]

    @staticmethod
    def get_volatility_features():
        """波动率因子（8个）"""
        return [
            # 价格波动率
            "Std($close / Ref($close, 1) - 1, 3)",          # 3日价格波动率
            "Std($close / Ref($close, 1) - 1, 5)",          # 5日价格波动率
            "Std($close / Ref($close, 1) - 1, 10)",         # 10日价格波动率
            "Std($close / Ref($close, 1) - 1, 20)",         # 20日价格波动率

            # 日内波动
            "Std($high - $low, 5) / Mean($close, 5)",       # 5日振幅标准差
            "Std($high - $low, 10) / Mean($close, 10)",     # 10日振幅标准差

            # ATR系列
            "Mean($high - $low, 14) / $close",               # 14日ATR（简化版）
            "Mean(Max($high - $low, Abs($high - Ref($close, 1)), Abs($low - Ref($close, 1))), 14) / $close",  # 完整ATR
        ]

    @staticmethod
    def get_technical_features():
        """技术指标因子（10个）"""
        return [
            # 价格区间
            "($high - $low) / $open",                        # 日内振幅
            "($high - $low) / $close",                       # 日内振幅（收盘价归一化）
            "(Max($high, 5) - Min($low, 5)) / Mean($close, 5)",  # 5日价格区间
            "(Max($high, 20) - Min($low, 20)) / Mean($close, 20)", # 20日价格区间

            # 价格位置
            "$close / $high",                                # 收盘价相对最高价
            "$close / $low",                                 # 收盘价相对最低价
            "($close - $low) / ($high - $low)",             # 收盘价在日内位置

            # 影线
            "($high - Max($open, $close)) / ($high - $low)", # 上影线比例
            "(Min($open, $close) - $low) / ($high - $low)",  # 下影线比例

            # 跳空
            "($open - Ref($close, 1)) / Ref($close, 1)",    # 跳空缺口
        ]

    @staticmethod
    def get_composite_features():
        """复合因子（10个）"""
        return [
            # 价量复合
            "($close / Ref($close, 1) - 1) * ($volume / Ref($volume, 1))",  # 价量联动
            "($close / Ref($close, 5) - 1) * ($volume / Mean($volume, 5))", # 5日价量联动
            "Corr($close, $volume, 10)",                     # 10日价量相关性

            # 波动率调整动量
            "($close / Ref($close, 5) - 1) / Std($close / Ref($close, 1) - 1, 5)",  # 波动率调整5日动量
            "($close / Ref($close, 10) - 1) / Std($close / Ref($close, 1) - 1, 10)", # 波动率调整10日动量

            # 成交量调整动量
            "($close / Ref($close, 5) - 1) * Mean($volume, 5) / $volume",   # 量调整5日动量
            "($close / Ref($close, 10) - 1) * Mean($volume, 10) / $volume", # 量调整10日动量

            # 价格范围调整
            "($close - Ref($close, 5)) / (Max($high, 5) - Min($low, 5))",  # 范围调整5日收益
            "($close - Ref($close, 10)) / (Max($high, 10) - Min($low, 10))", # 范围调整10日收益

            # 指数加权
            "(Ref($close, 1) * 0.5 + Ref($close, 2) * 0.3 + Ref($close, 3) * 0.2) / $close - 1",  # 指数加权动量
        ]

    @staticmethod
    def get_all_features():
        """获取所有RD-Agent因子"""
        features = []
        features.extend(RDAgentFactors.get_price_momentum_features())
        features.extend(RDAgentFactors.get_volume_features())
        features.extend(RDAgentFactors.get_volatility_features())
        features.extend(RDAgentFactors.get_technical_features())
        features.extend(RDAgentFactors.get_composite_features())
        return features

    @staticmethod
    def get_feature_count():
        """获取因子数量统计"""
        return {
            "price_momentum": len(RDAgentFactors.get_price_momentum_features()),
            "volume": len(RDAgentFactors.get_volume_features()),
            "volatility": len(RDAgentFactors.get_volatility_features()),
            "technical": len(RDAgentFactors.get_technical_features()),
            "composite": len(RDAgentFactors.get_composite_features()),
            "total": len(RDAgentFactors.get_all_features()),
        }
```

**因子统计**:
- 价格动量: 12个
- 成交量: 10个
- 波动率: 8个
- 技术指标: 10个
- 复合因子: 10个
- **总计: 50个高质量因子**
